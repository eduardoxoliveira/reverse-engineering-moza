# =============================================================================
# ENGENHARIA REVERSA MOZA — V8: frida-trace CLI (mais simples e robusta)
# =============================================================================
#
# A V7 usou frida.spawn via API Python — pode ter tido problema de bitness
# (Python 64-bit x FirmwareManager 32-bit).
#
# V8 usa `frida-trace` CLI, que:
#   - Detecta bitness automaticamente
#   - Instala/injeta agent correto
#   - Cria arquivos "handler" em __handlers__/crypt.dll/ que voce pode editar
#   - Loga cada call em tempo real, sem precisar codar hooks
#
# ANTES DE RODAR:
#   1. Feche qualquer FirmwareManager.exe ja aberto
#   2. Conecte o volante FSR1 (opcional)
# =============================================================================

python -m pip install frida-tools

$logPath = "$env:USERPROFILE\Desktop\moza-re\output_v8.txt"
Start-Transcript -Path $logPath -Force -IncludeInvocationHeader | Out-Null
Write-Host "`n=== Log em: $logPath ===`n"

$env:QT_LOGGING_RULES = "*=true"
$env:QT_FORCE_STDERR_LOGGING = "1"

# Criar diretorio de trabalho pra frida-trace
$traceDir = "$env:USERPROFILE\Desktop\moza-re\frida-trace"
New-Item -ItemType Directory -Force -Path $traceDir | Out-Null
Set-Location $traceDir

Write-Host "=== INICIANDO frida-trace ===" -ForegroundColor Cyan
Write-Host "Isso vai:" -ForegroundColor Yellow
Write-Host " 1. Spawnar FirmwareManager.exe" -ForegroundColor Yellow
Write-Host " 2. Injetar tracers para Tea_Init, Tea_Encrypt, Tea_Decrypt" -ForegroundColor Yellow
Write-Host " 3. Loggar cada chamada em tempo real" -ForegroundColor Yellow
Write-Host ""
Write-Host "Handlers gerados em: $traceDir\__handlers__\crypt.dll\" -ForegroundColor Yellow
Write-Host ""
Write-Host "ACAO DO USUARIO:" -ForegroundColor Cyan
Write-Host " - A UI do FirmwareManager vai abrir"
Write-Host " - Selecione o volante FSR1"
Write-Host " - Clique em 'Update Firmware' ou 'Verificar Update'"
Write-Host " - Se pedir confirmacao pra flashear, CANCELE"
Write-Host " - Se aparecer '[Tea_Init] chamada', copia os hex bytes"
Write-Host ""
Write-Host "Ctrl+C aqui para encerrar."
Write-Host ""

# frida-trace principal — spawna FirmwareManager com tracers em crypt.dll
$fmw = "C:\Program Files (x86)\MOZA Pit House\bin\FirmwareManager.exe"
frida-trace -f "$fmw" -i "crypt.dll!Tea_Init" -i "crypt.dll!Tea_Encrypt" -i "crypt.dll!Tea_Decrypt"

# Se sair automaticamente sem capturar nada, ver arquivos handler
Write-Host "`n=== Handlers gerados ===" -ForegroundColor Cyan
Get-ChildItem "$traceDir\__handlers__" -Recurse -File | ForEach-Object {
    Write-Host "`n--- $($_.FullName) ---" -ForegroundColor Yellow
    Get-Content $_.FullName
}

Stop-Transcript | Out-Null

# Limpar env vars
Remove-Item Env:QT_LOGGING_RULES -ErrorAction SilentlyContinue
Remove-Item Env:QT_FORCE_STDERR_LOGGING -ErrorAction SilentlyContinue

Write-Host "`n=== Log em: $logPath ===" -ForegroundColor Green
